"use strict";
var Store = /** @class */ (function () {
    function Store() {
    }
    /**
     *
     * @author chwech
     * @date 2019-10-26
     * @param {string} name 缓存的键名，支持"wechat.user.name"格式来获取值，前提是缓存值是一个JSON字符串
     * @memberof Store
     */
    Store.prototype.getStore = function (name) {
        var timestamp = Date.parse(new Date().toUTCString()) / 1000; // 时间戳 秒为单位
        if (!name)
            return;
        var key = name.split('.')[0], value = '';
        if (window.localStorage) {
            var strVal = window.localStorage.getItem(key);
            value = window.localStorage.getItem(key) === null ? '' : strVal;
        }
        try {
            var val_1 = JSON.parse(value);
            if (val_1._endtime && timestamp > val_1._endtime)
                return; // 已过期
            if (!val_1.hasOwnProperty('data'))
                return;
            val_1 = val_1.data;
            name.split('.').forEach(function (segment, index) {
                if (index > 0) {
                    if (val_1.hasOwnProperty(segment)) {
                        val_1 = val_1[segment];
                    }
                    else {
                        val_1 = false;
                    }
                }
            });
            return val_1;
        }
        catch (e) {
            return false;
        }
    };
    return Store;
}());
